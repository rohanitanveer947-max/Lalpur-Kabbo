import os
import logging
import requests
from flask import Flask, request, jsonify
from openai import OpenAI
from dotenv import load_dotenv

# এনভায়রনমেন্ট ভেরিয়েবল লোড করা
load_dotenv()
app = Flask(__name__)

# কনফিগারেশন
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
ACCESS_TOKEN = os.getenv("FACEBOOK_PAGE_ACCESS_TOKEN")
VERIFY_TOKEN = os.getenv("VERIFY_TOKEN")
SYSTEM_PROMPT = "তুমি রোহানি তানভীরের এআই প্রতিনিধি। তুমি মার্জিত, বুদ্ধিদীপ্ত এবং সহযোগিতামূলক।"

# লগিং সেটআপ (ত্রুটি ট্র্যাক করার জন্য)
logging.basicConfig(level=logging.INFO)

# এআই এর সাথে কথোপকথন পরিচালনা
def get_ai_response(user_input, conversation_history):
    try:
        messages = [{"role": "system", "content": SYSTEM_PROMPT}] + conversation_history
        messages.append({"role": "user", "content": user_input})
        
        response = client.chat.completions.create(
            model="gpt-4o", # বা gpt-3.5-turbo
            messages=messages
        )
        return response.choices[0].message.content
    except Exception as e:
        logging.error(f"AI Error: {e}")
        return "দুঃখিত, এই মুহূর্তে আমি আপনার উত্তরের জন্য প্রস্তুত নই।"

# ফেসবুকে রিপ্লাই পাঠানোর ফাংশন
def send_facebook_reply(comment_id, message):
    url = f"https://graph.facebook.com/v20.0/{comment_id}/comments"
    params = {"access_token": ACCESS_TOKEN, "message": message}
    response = requests.post(url, params=params)
    return response.status_code

@app.route('/webhook', methods=['GET', 'POST'])
def webhook():
    if request.method == 'GET':
        if request.args.get("hub.verify_token") == VERIFY_TOKEN:
            return request.args.get("hub.challenge")
        return "Verification Failed", 403

    data = request.json
    logging.info(f"Received data: {data}")

    try:
        if "entry" in data:
            for entry in data['entry']:
                for change in entry.get('changes', []):
                    value = change.get('value', {})
                    if value.get('verb') == 'add': # নতুন কমেন্ট হলে
                        comment_id = value.get('comment_id')
                        user_message = value.get('message')
                        
                        # এআই থেকে উত্তর নিন
                        reply = get_ai_response(user_message, [])
                        
                        # ফেসবুকে পাঠান
                        send_facebook_reply(comment_id, reply)
    except Exception as e:
        logging.error(f"Webhook Processing Error: {e}")

    return jsonify({"status": "success"}), 200

if __name__ == '__main__':
    # প্রোডাকশনের জন্য gunicorn বা uWSGI ব্যবহার করুন
    app.run(port=5000, debug=False)
